// --- WASM/Emscripten main loop (bgfx) ---
// Mirrors the raylib wasm template shape, adapted to bgfx's window API. bgfx's
// web target drives the frame through `emscripten_set_main_loop` (a per-frame
// callback) instead of the desktop `while (!window.shouldQuit())` loop: the
// browser owns the event loop, so a blocking while-loop would freeze the page.
// The callback body mirrors the bgfx DESKTOP template's loop body
// (tick -> beginFrame -> render -> endFrame), minus the desktop-only
// headless / screenshot / fullscreen / vsync plumbing (none of which applies in
// the browser).
//
// The bgfx WebGL context is created against the emscripten `#canvas` element:
// `window.initWindow(w, h, title)` comptime-selects `initWindowWasm`, which hands
// bgfx the canvas selector via `PlatformData.nwh` (see the backend's
// src/window.zig). `beginFrame`/`clearBackground`/`render`/`endFrame` are the same
// backend-agnostic gfx seam the desktop + android templates use.
//
// Editor-preview holes (labelle-studio Play mode, assembler ≥ 0.74.0):
// `editor_sim_open` / `editor_sim_close` bracket the SIM half of
// `gameFrame` (tick block + `g.tick(dt)`; render stays unconditional) and
// `editor_bind` sits between the setup code and the
// `emscripten_set_main_loop` call. A NON-preview generation fills all three
// EMPTY — the hole placement is chosen so that empty fill reproduces the
// pre-preview template bytes exactly. NOTE: an assembler that predates the
// holes (< 0.74.0) leaves them verbatim in the generated main.zig (a compile
// error) — this template version must be paired with assembler ≥ 0.74.0.

// ── Zig 0.16 emscripten std shims (bgfx-wasm labelle-bgfx#8) ────────────────
// Zig 0.16.0's DEFAULT panic + log handlers reach `std.Io.Threaded`'s
// child-process wait code (via the debug stack-trace printer), which does NOT
// compile for wasm32-emscripten (a std regression: `posix.W.STOPSIG` returns
// `u32` but `statusToTerm` expects the SIG enum). Overriding BOTH the panic entry
// (trap) and the log fn (route to the browser console, formatting into a stack
// buffer — never `debug_io`) severs that reference so the browser build links.
extern "c" fn emscripten_console_log(str: [*:0]const u8) void;

pub const std_options: std.Options = .{
    .logFn = struct {
        fn logFn(
            comptime level: std.log.Level,
            comptime scope: @TypeOf(.enum_literal),
            comptime format: []const u8,
            args: anytype,
        ) void {
            _ = scope;
            var buf: [512]u8 = undefined;
            const line = std.fmt.bufPrintZ(&buf, "[" ++ level.asText() ++ "] " ++ format, args) catch return;
            emscripten_console_log(line.ptr);
        }
    }.logFn,
};

pub const panic = std.debug.FullPanic(struct {
    fn handler(msg: []const u8, ret_addr: ?usize) noreturn {
        _ = msg;
        _ = ret_addr;
        @trap();
    }
}.handler);

fn gameFrame() callconv(.c) void {
    // Frame-rate-independent sim: derive dt from the real frame period
    // (`window.frameDuration` measures it with a monotonic clock). Clamp to 4
    // frames at target_fps so a stall / backgrounded tab can't explode physics.
    // @max(target_fps, 1) guards a target_fps=0 (unlimited) config. Mirrors the
    // desktop template.
    const dt: f32 = @min(@as(f32, @floatCast(window.frameDuration())), 4.0 / @as(f32, @floatFromInt(@max(target_fps, 1))));

    // Aspect-fit the design canvas (config) into the real canvas size, so
    // content renders correctly regardless of the browser canvas dimensions.
    BackendGfxType.setScreenSize(window.width(), window.height());
    BackendGfxType.setDesignSize(@intCast(screen_w), @intCast(screen_h));

// Hand-rolled extern shim instead of `@cImport(emscripten.h)` — Zig 0.16's
// translate-c rejects recent emsdk headers (multi-arg

// need the main-loop entry; emcc resolves the symbol at link time. Mirrors the

pub fn main() !void {
    // Use c_allocator for Emscripten — it delegates to emscripten's malloc/free
    // which respects ALLOW_MEMORY_GROWTH. page_allocator uses wasm memory.grow
    // directly and conflicts with emscripten's memory management. Mirrors the
    // raylib wasm template.
    const allocator = std.heap.c_allocator;
    g_allocator = allocator;

{{hidden_setup}}    window.initWindow(screen_w, screen_h, screen_title);
    // Note: no `defer window.closeWindow()` — Emscripten keeps running after
    // main returns (the main-loop callback drives the app).

    g = AssembledGame.init(allocator);

    g.setHooks(&hooks);

    g.setScreenHeight(@as(f32, @floatFromInt(screen_h)));

    // Script Runtime Contract bind (labelle-engine#749 / labelle-assembler#593):
    // hand the game to the engine's script_contract once, right after Game.init
    // and BEFORE the assembler's setup code below runs the scripting plugin's
    // setup (which boots the VM against the exported C surface). Double
    // @hasDecl gate — engine exposes the contract AND the assembler generated
    // the `scripting_enabled` flag — mirroring the desktop template (and the
    // fullscreen/vsync drain convention), so older engines and script-less
    // projects build unchanged. The per-frame twin,
    // `script_contract.drainEvents`, rides the assembler's tick_code
    // splice in gameFrame — between the plugin ticks and g.dispatchEvents().
    if (comptime @hasDecl(engine, "script_contract") and @hasDecl(@This(), "scripting_enabled")) {
        engine.script_contract.bind(&g);
        // Paired teardown BEFORE g.deinit(): the contract owns
        // subscription/inbox/emit-arena state through g.allocator and
        // must release it (and drop the game pointer) while the game
        // is still alive — otherwise the debug allocator reports the
        // contract's state as leaked and the exports stay bound to a
        // dying game.
    }
    defer if (comptime @hasDecl(engine, "script_contract") and @hasDecl(@This(), "scripting_enabled")) {
        engine.script_contract.unbind();
    };
