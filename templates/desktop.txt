        // screenshot is still pending (`screenshot_req != null`, so `--ticks`
        // can't cut off a capture whose `--after` delay hasn't elapsed) or

        if (headless_max_ticks != 0 and headless_tick_count >= headless_max_ticks and !screenshot_blocks_exit) {

        }
